Task 3: Drawing Application

Form1.cs:
```
using System;
using System.Drawing;
using System.Windows.Forms;

namespace DrawingApplication
{
    public partial class Form1 : Form
    {
        private bool drawing = false;
        private Point lastPoint;

        public Form1()
        {
            InitializeComponent();
        }

        private void Form1_MouseDown(object sender, MouseEventArgs e)
        {
            drawing = true;
            lastPoint = e.Location;
        }

        private void Form1_MouseMove(object sender, MouseEventArgs e)
        {
            if (drawing)
            {
                Graphics g = CreateGraphics();
                g.DrawLine(Pens.Black, lastPoint, e.Location);
                lastPoint = e.Location;
            }
        }

        private void Form1_MouseUp(object sender, MouseEventArgs e)
        {
            drawing = false;
        }
    }
}
```

This solution creates a Windows Forms application that allows users to draw on the form using the mouse. The `Form1_MouseDown`, `Form1_MouseMove`, and `Form1_MouseUp` event handlers capture the mouse events to draw lines on the form.
